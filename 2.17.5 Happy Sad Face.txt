def smile_mouth():
    pendown()
    pensize(6)
    forward(4)
    left(3)
def frown_mouth():
    pendown()
    pensize(6)
    forward(-4)
    left(-3)
happy = input("happy or not?")
if happy == "happy":
    penup()
    setposition(0,-100)
    pendown()
